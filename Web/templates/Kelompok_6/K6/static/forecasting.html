def clean_text(text):
    text = str(text).lower()
    text = re.sub(r"http\S+", " ", text)
    text = re.sub(r"www\S+", " ", text)
    text = re.sub(r"@\w+", " ", text)
    text = re.sub(r"#", " ", text)
    text = re.sub(r"[^a-zA-Z\s]", " ", text)
    text = re.sub(r"\s+", " ", text)
    text = text.strip()
    text = normalize_slang(text)
    return text

def synonym_replacement(text):
    words = text.split()
    result = []
    for word in words:
        if word in synonym_dict:
            if random.random() < 0.5:
                word = random.choice(
                    synonym_dict[word]
                )
        result.append(word)
    return " ".join(result)

class SentimentDataset(
    torch.utils.data.Dataset
):

    def __init__(
        self,
        encodings,
        labels
    ):
        self.encodings = encodings
        self.labels = labels

    def __getitem__(
        self,
        idx
    ):

        item = {
            key: torch.tensor(
                val[idx]
            )
            for key, val
            in self.encodings.items()
        }

        item["labels"] = torch.tensor(
            self.labels[idx]
        )

        return item

    def __len__(self):
        return len(
            self.labels
        )

def parse_relative_time(time_str, base_date=None):

    if base_date is None:
        base_date = pd.Timestamp.now()

    text = str(time_str).lower().replace("(edited)", "").strip()

    patterns = [
        (r"(\d+)\s*year",   lambda n: base_date - pd.DateOffset(years=int(n))),
        (r"(\d+)\s*month",  lambda n: base_date - pd.DateOffset(months=int(n))),
        (r"(\d+)\s*week",   lambda n: base_date - pd.DateOffset(weeks=int(n))),
        (r"(\d+)\s*day",    lambda n: base_date - pd.DateOffset(days=int(n))),
        (r"(\d+)\s*hour",   lambda n: base_date - pd.DateOffset(hours=int(n))),
        (r"(\d+)\s*minute", lambda n: base_date - pd.DateOffset(minutes=int(n))),
    ]

    for pattern, fn in patterns:
        m = re.search(pattern, text)
        if m:
            return fn(m.group(1))

    return pd.NaT

def forecast_recursive(model, last_seq, steps):
    curr_seq = last_seq.copy()
    preds = []
    for _ in range(steps):
        inp = curr_seq.reshape(1, LOOK_BACK, 1)
        p = model.predict(inp, verbose=0)[0,0]
        preds.append(p)
        curr_seq = np.append(curr_seq[1:], p)
    return np.array(preds)
